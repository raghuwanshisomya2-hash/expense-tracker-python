# Expense Tracker Application

def add_expense():
    date = input("Enter date (DD-MM-YYYY): ")
    category = input("Enter category (Food, Travel, Study, etc): ")
    amount = input("Enter amount: ")
    description = input("Enter description: ")

    with open("expenses.txt", "a") as file:
        file.write(f"{date},{category},{amount},{description}\n")

    print("Expense added successfully!\n")


def view_expenses():
    try:
        with open("expenses.txt", "r") as file:
            print("\nDate | Category | Amount | Description")
            print("-" * 40)
            for line in file:
                date, category, amount, description = line.strip().split(",")
                print(f"{date} | {category} | {amount} | {description}")
            print()
    except FileNotFoundError:
        print("No expenses found.\n")


def total_expense():
    total = 0
    try:
        with open("expenses.txt", "r") as file:
            for line in file:
                total += float(line.strip().split(",")[2])
        print(f"\nTotal Expense: ₹{total}\n")
    except FileNotFoundError:
        print("No expenses found.\n")


def category_summary():
    summary = {}
    try:
        with open("expenses.txt", "r") as file:
            for line in file:
                category = line.strip().split(",")[1]
                amount = float(line.strip().split(",")[2])
                summary[category] = summary.get(category, 0) + amount

        print("\nCategory-wise Expense Summary")
        print("-" * 35)
        for category, amount in summary.items():
            print(f"{category}: ₹{amount}")
        print()
    except FileNotFoundError:
        print("No expenses found.\n")


while True:
    print("------ Expense Tracker ------")
    print("1. Add Expense")
    print("2. View Expenses")
    print("3. Total Expense")
    print("4. Category-wise Summary")
    print("5. Exit")

    choice = input("Enter your choice (1-5): ")

    if choice == "1":
        add_expense()
    elif choice == "2":
        view_expenses()
    elif choice == "3":
        total_expense()
    elif choice == "4":
        category_summary()
    elif choice == "5":
        print("Thank you for using Expense Tracker!")
        break
    else:
        print("Invalid choice. Try again.\n")
